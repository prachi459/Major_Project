    public void login()
    {
        String username = txtuser.getText();
        String password = txtpass.getText();
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            con = DriverManager.getConnection("jdbc:mysql://localhost/superposs","root","");
            pst = con.prepareStatement("select * from cashier where username =? and password =?");
            
            pst.setString(1, username);
            pst.setString(2, password);
            rs = pst.executeQuery();
            if(rs.next())
            {
                pos a = new pos();
                this.hide();
                new pos(txtuser.getText()).setVisible(true);
            }
             else
            {
                JOptionPane.showMessageDialog(this,"Username and Password do not Match");
                txtuser.setText("");
                txtuser.requestFocus();
            }
        } catch (ClassNotFoundException ex) {
            Logger.getLogger(login.class.getName()).log(Level.SEVERE, null, ex);
        } catch (SQLException ex) {
            Logger.getLogger(login.class.getName()).log(Level.SEVERE, null, ex);
        }
      
    }
